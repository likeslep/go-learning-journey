# gorm


## 分页函数
```go
func Paginate(r *http.Request) func(db *gorm.DB) *gorm.DB {
  return func (db *gorm.DB) *gorm.DB {
    q := r.URL.Query()
    page, _ := strconv.Atoi(q.Get("page"))
    if page <= 0 {
      page = 1
    }

    pageSize, _ := strconv.Atoi(q.Get("page_size"))
    switch {
    case pageSize > 100:
      pageSize = 100
    case pageSize <= 0:
      pageSize = 10
    }

    offset := (page - 1) * pageSize
    return db.Offset(offset).Limit(pageSize)
  }
}
```

本质是调用gorm的Offset和Limit()方法。


## Scopes()方法

Scopes()是GORM提供的一个强大功能，用于链式复用查询条件，可以将常用的查询逻辑封装成可重用的函数。Scopes()允许传入一个或多个函数，这些函数会修改当前的查询对象，实现查询条件的动态组合。

```go
type Scope func(*gorm.DB) *gorm.DB

db.Scopes(scope1, scope2, ...).Find(&users)
```

1. 封装常用查询条件
```go
// 定义 Scope 函数
func AgeGreaterThan(age int) func(db *gorm.DB) *gorm.DB {
    return func(db *gorm.DB) *gorm.DB {
        return db.Where("age > ?", age)
    }
}

func StatusActive(db *gorm.DB) *gorm.DB {
    return db.Where("status = ?", "active")
}

func RecentOrder(db *gorm.DB) *gorm.DB {
    return db.Order("created_at DESC")
}

// 使用 Scopes
var users []User
db.Scopes(StatusActive, AgeGreaterThan(18), RecentOrder).Find(&users)
```

2. 分页封装
```go
// 分页 Scope
func Paginate(page, pageSize int) func(db *gorm.DB) *gorm.DB {
    return func(db *gorm.DB) *gorm.DB {
        offset := (page - 1) * pageSize
        return db.Offset(offset).Limit(pageSize)
    }
}

// 使用分页
var users []User
db.Scopes(Paginate(1, 10)).Find(&users)
```

3. 软删除查询
```go
// 包含已删除的记录
func Unscoped(db *gorm.DB) *gorm.DB {
    return db.Unscoped()
}

// 只查询已删除的记录
func OnlyDeleted(db *gorm.DB) *gorm.DB {
    return db.Unscoped().Where("deleted_at IS NOT NULL")
}

// 使用
var deletedUsers []User
db.Scopes(OnlyDeleted).Find(&deletedUsers)
```
